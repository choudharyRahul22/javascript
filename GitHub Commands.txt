// To clone a repo from github to local system
git clone https://github.com/choudharyRahul22/javascript

// move to github repo
cd javascript

//will show the current status what is in repo and in local system
git status

// will add the file to repo
git add javascript.txt

// add all files
git add -A 

// will lock the file for adding
git commit -m "javascript.txt file"

// will add the file to repo
git push
